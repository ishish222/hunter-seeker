#!/bin/bash

ADDR=10.8.0.6

#remove old if exists
ssh hs1@$ADDR "if [ -f /tmp/patch ]; then patch -d /home/hs1/hs-current/ -Rp1 < /tmp/patch && rm /tmp/patch; fi"

hg diff > /tmp/patch
scp /tmp/patch hs1@$ADDR:/tmp/patch
#install new
ssh hs1@$ADDR patch -d /home/hs1/hs-current/ -p1 < /tmp/patch
ssh hs1@$ADDR cp -r /home/hs1/hs/hs-0.2/origins /home/hs1/hs-current/
ssh hs1@$ADDR cp -r /home/hs1/config/.default.* /home/hs1/hs-current/src/
